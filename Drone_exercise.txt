public void register_drone(char[100] pSerial, Model pModel, float pWeight, float pBattery,State pState, ArrayList<Medication> pL_Medication )
	{
		Drone drone = new Drone();
		drone.serial_number = pSerial;
		drone.model = pModel;
		drone.weight = pWeight;
		drone.battery_cap = pBattery;
		drone.State = pState;
		drone.medication = pL_Medication;
				
	}

public void load_Drone ()
	{
		Drone drone = new Drone();
		Medication medicate = new Medication();
		if(drone.state == IDLE)
		{
			for (int i=0; i < drone.medication_List.size(); i++)
			{
				drone.set_state(LOADING);
				if (drone.weight < 500 && drone.battery_cap > 25)
				{
					medicate = drone.medication_List.get(i);
					if(medicate.weight < 500)
					{
						drone.medication_List.add(medicate);
						drone.weight += medicate.weight;
					}
				}
			}		
			
		}
		else
		System.out.println("The drone could not load");
	
	drone.set_state(LOADED);
	//save task
	}	
public void check_Medications(Drone pDrone)
	{
		load_Drone();
